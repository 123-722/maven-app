pipeline {
    agent any
    stages {
        stage ('git checkout') {
            steps{
               git url: 'https://github.com/mohandevops1869/simple-java-maven-app.git'
            }   
        }
        stage ('maven build') {
            steps {
                script {
                    sh """
                     export MAVEN_HOME=/opt/maven
                     export PATH=$PATH:$MAVEN_HOME/bin
                     mvn --version
                     mvn clean package  
                    """
                }
            }
        }
    }
}
